13/// GPU-accelerated gradient computation for neural networks
14pub struct GpuGradientComputer<T: Float + Send + Sync + Default + std::fmt::Debug + 'static> {
15    backend: Arc<dyn ComputeBackend<T>>,
16    _phantom: std::marker::PhantomData<T>,
17}
18
19impl<T: Float + Send + Sync + Default + std::fmt::Debug + 'static> GpuGradientComputer<T> {
20    /// Create a new GPU gradient computer
21    pub fn new(backend: Arc<dyn ComputeBackend<T>>) -> Self {
22        Self {
23            backend,
24            _phantom: std::marker::PhantomData,
25        }
26    }
27
28    /// Perform full backpropagation on GPU
29    pub fn backpropagate(
30        &self,
31        network: &Network<T>,
32        layer_activations: &[Vec<T>],
33        output_error: &[T],
34    ) -> Result<(Vec<Vec<T>>, Vec<Vec<T>>), ComputeError> {
35        let mut weight_gradients = Vec::new();
36        let mut bias_gradients = Vec::new();
37        let mut current_error = output_error.to_vec();
38
39        // Process layers in reverse order (backpropagation)
40        for layer_idx in (1..network.layers.len()).rev() {
41            let layer = &network.layers[layer_idx];
42            let prev_layer = &network.layers[layer_idx - 1];
43
44            // Get activations from previous layer
45            let prev_activations = &layer_activations[layer_idx - 1];
46
47            // Extract layer dimensions
48            let output_size = layer.neurons.iter().filter(|n| !n.is_bias).count();
49            let input_size = prev_activations.len();
50
51            // Compute gradients for this layer
52            let (layer_weight_grad, layer_bias_grad, prev_layer_error) = self
53                .compute_layer_gradients(
54                    layer,
55                    prev_activations,
56                    &current_error,
57                    input_size,
58                    output_size,
59                )?;
60
61            // Store gradients (in forward order)
62            weight_gradients.insert(0, layer_weight_grad);
63            bias_gradients.insert(0, layer_bias_grad);
64
65            // Update error for next iteration (previous layer)
66            if layer_idx > 1 {
67                current_error = prev_layer_error;
68            }
69        }
70
71        Ok((weight_gradients, bias_gradients))
72    }
73
74    /// Compute gradients for a single layer using GPU
75    fn compute_layer_gradients(
76        &self,
77        layer: &crate::Layer<T>,
78        prev_activations: &[T],
79        layer_error: &[T],
80        input_size: usize,
81        output_size: usize,
82    ) -> Result<(Vec<T>, Vec<T>, Vec<T>), ComputeError> {
83        // Extract weights matrix for this layer
84        let weights = self.extract_weights_matrix(layer, input_size, output_size);
85
86        // Compute weight gradients: dW = prev_activations^T * layer_error
87        // For each weight w_ij: dw_ij = activation_i * error_j
88        let mut weight_gradients = Vec::with_capacity(weights.len());
89        for output_idx in 0..output_size {
90            for input_idx in 0..input_size {
91                let gradient = prev_activations[input_idx] * layer_error[output_idx];
92                weight_gradients.push(gradient);
93            }
94        }
95
96        // Bias gradients are simply the error terms
97        let bias_gradients = layer_error.to_vec();
98
99        // Compute error for previous layer: prev_error = weights^T * layer_error
100        let prev_layer_error = if input_size > 0 {
101            self.backend.matrix_vector_multiply(
102                &self.transpose_weights(&weights, input_size, output_size),
103                layer_error,
104                input_size,
105                output_size,
106            )?
107        } else {
108            vec![T::zero(); input_size]
109        };
110
111        // Apply activation function derivative to previous layer error
112        let prev_layer_error = self.apply_activation_derivative(
113            &prev_layer_error,
114            prev_activations,
115            layer.neurons[0].activation_function,
116        )?;
117
118        Ok((weight_gradients, bias_gradients, prev_layer_error))
119    }

121    /// Extract weights as a matrix from a layer
122    fn extract_weights_matrix(
123        &self,
124        layer: &crate::Layer<T>,
125        input_size: usize,
126        output_size: usize,
127    ) -> Vec<T> {
128        let mut weights = vec![T::zero(); output_size * input_size];
129
130        for (neuron_idx, neuron) in layer.neurons.iter().filter(|n| !n.is_bias).enumerate() {
131            // Skip bias connection (first connection)
132            for (conn_idx, connection) in neuron.connections.iter().skip(1).enumerate() {
133                if conn_idx < input_size && neuron_idx < output_size {
134                    weights[neuron_idx * input_size + conn_idx] = connection.weight;
135                }
136            }
137        }
138
139        weights
140    }
141
142    /// Transpose weight matrix
143    fn transpose_weights(&self, weights: &[T], rows: usize, cols: usize) -> Vec<T> {
144        let mut transposed = vec![T::zero(); rows * cols];
145
146        for row in 0..rows {
147            for col in 0..cols {
148                transposed[col * rows + row] = weights[row * cols + col];
149            }
150        }
151
152        transposed
153    }
154
155    /// Apply activation function derivative
156    fn apply_activation_derivative(
157        &self,
158        errors: &[T],
159        activations: &[T],
160        activation_fn: crate::ActivationFunction,
161    ) -> Result<Vec<T>, ComputeError> {
162        use crate::ActivationFunction::*;
163
164        let mut result = Vec::with_capacity(errors.len());
165
166        match activation_fn {
167            Sigmoid => {
168                // f'(x) = f(x) * (1 - f(x))
169                for (i, &error) in errors.iter().enumerate() {
170                    let activation = activations[i];
171                    let derivative = activation * (T::one() - activation);
172                    result.push(error * derivative);
173                }
174            }
175            Tanh => {
176                // f'(x) = 1 - f(x)^2
177                for (i, &error) in errors.iter().enumerate() {
178                    let activation = activations[i];
179                    let derivative = T::one() - activation * activation;
180                    result.push(error * derivative);
181                }
182            }
183            ReLU => {
184                // f'(x) = 1 if x > 0, 0 otherwise
185                for (i, &error) in errors.iter().enumerate() {
186                    let derivative = if activations[i] > T::zero() {
187                        T::one()
188                    } else {
189                        T::zero()
190                    };
191                    result.push(error * derivative);
192                }
193            }
194            Linear => {
195                // f'(x) = 1
196                result = errors.to_vec();
197            }
198            _ => {
199                // For unsupported activation functions, fall back to sigmoid derivative
200                for (i, &error) in errors.iter().enumerate() {
201                    let activation = activations[i];
202                    let derivative = activation * (T::one() - activation);
203                    result.push(error * derivative);
204                }
205            }
206        }
207
208        Ok(result)
209    }
210}

212/// GPU-accelerated forward propagation
213pub fn gpu_forward_propagate<T: Float + Send + Sync + Default + std::fmt::Debug + 'static>(
214    backend: &Arc<dyn ComputeBackend<T>>,
215    network: &Network<T>,
216    input: &[T],
217) -> Result<Vec<Vec<T>>, ComputeError> {
218    let mut activations = vec![input.to_vec()];
219    let mut current_input = input.to_vec();
220
221    // Process each layer
222    for layer_idx in 1..network.layers.len() {
223        let layer = &network.layers[layer_idx];
224        let output_size = layer.neurons.iter().filter(|n| !n.is_bias).count();
225        let input_size = current_input.len();
226
227        // Extract weights and biases
228        let mut weights = Vec::new();
229        let mut biases = Vec::new();
230
231        for neuron in layer.neurons.iter().filter(|n| !n.is_bias) {
232            // Extract bias (first connection)
233            let bias = if !neuron.connections.is_empty() {
234                neuron.connections[0].weight
235            } else {
236                T::zero()
237            };
238            biases.push(bias);
239
240            // Extract weights (skip bias connection)
241            for connection in neuron.connections.iter().skip(1) {
242                weights.push(connection.weight);
243            }
244        }
245
246        // Perform matrix-vector multiplication on GPU
247        let layer_output =
248            backend.matrix_vector_multiply(&weights, &current_input, output_size, input_size)?;
249
250        // Add biases
251        let mut with_bias = Vec::with_capacity(layer_output.len());
252        for (i, &output) in layer_output.iter().enumerate() {
253            with_bias.push(output + biases[i]);
254        }
255
256        // Apply activation function
257        let activation_fn = layer
258            .neurons
259            .iter()
260            .find(|n| !n.is_bias)
261            .map(|n| n.activation_function)
262            .unwrap_or(crate::ActivationFunction::Sigmoid);
263
264        let activated = backend.apply_activation_function(
265            &with_bias,
266            activation_fn,
267            T::one(), // steepness
268        )?;
269
270        activations.push(activated.clone());
271        current_input = activated;
272    }
273
274    Ok(activations)
275}
