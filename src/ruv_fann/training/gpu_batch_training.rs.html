13/// Batch-optimized GPU training implementation
14pub struct BatchGpuTrainer<T: Float + Send + Sync + Default + std::fmt::Debug + 'static> {
15    backend: Arc<dyn ComputeBackend<T>>,
16    batch_size: usize,
17    _phantom: std::marker::PhantomData<T>,
18}

20impl<T: Float + Send + Sync + Default + std::fmt::Debug + 'static> BatchGpuTrainer<T> {
21    /// Create a new batch GPU trainer
22    pub fn new(backend: Arc<dyn ComputeBackend<T>>, batch_size: usize) -> Self {
23        Self {
24            backend,
25            batch_size,
26            _phantom: std::marker::PhantomData,
27        }
28    }

30    /// Process entire batch in a single GPU operation
31    /// This implementation uses GPU shaders for maximum performance
32    pub fn batch_forward_pass(
33        &self,
34        network: &Network<T>,
35        batch_inputs: &[Vec<T>],
36    ) -> Result<Vec<Vec<T>>, ComputeError> {
37        let batch_size = batch_inputs.len();
38        let mut all_activations = vec![batch_inputs.to_vec()];
39
40        // Process through each layer using batch matrix multiplication
41        for (layer_idx, layer) in network.layers.iter().skip(1).enumerate() {
42            let prev_activations = &all_activations.last().unwrap();
43            let current_layer_size = layer.neurons.iter().filter(|n| !n.is_bias).count();
44
45            // Extract weights and biases once for the entire batch
46            let (weight_matrix, biases) =
47                self.extract_layer_parameters(layer, prev_activations[0].len(), current_layer_size);
48
49            // Use GPU batch matrix multiplication - this uses the batch_matrix_vector_multiply.wgsl shader
50            let batch_outputs = self.backend.batch_matrix_vector_multiply(
51                &weight_matrix,
52                prev_activations,
53                current_layer_size,
54                prev_activations[0].len(),
55            )?;
56
57            // Add biases and apply activation function on GPU
58            let mut activated_outputs = Vec::with_capacity(batch_size);
59            for mut output in batch_outputs {
60                // Add biases
61                for (i, bias) in biases.iter().enumerate() {
62                    output[i] = output[i] + *bias;
63                }
64
65                // Apply activation function using GPU shader
66                let activation_fn = layer
67                    .neurons
68                    .iter()
69                    .find(|n| !n.is_bias)
70                    .map(|n| n.activation_function)
71                    .unwrap_or(crate::ActivationFunction::Sigmoid);
72
73                let activated = self.backend.apply_activation_function(
74                    &output,
75                    activation_fn,
76                    T::one(), // steepness
77                )?;
78
79                activated_outputs.push(activated);
80            }
81
82            all_activations.push(activated_outputs);
83        }
84
85        Ok(all_activations.last().unwrap().clone())
86    }

88    /// Extract weights and biases from a layer
89    fn extract_layer_parameters(
90        &self,
91        layer: &crate::Layer<T>,
92        input_size: usize,
93        output_size: usize,
94    ) -> (Vec<T>, Vec<T>) {
95        let mut weights = Vec::with_capacity(output_size * input_size);
96        let mut biases = Vec::with_capacity(output_size);
97
98        for neuron in layer.neurons.iter().filter(|n| !n.is_bias) {
99            // Extract bias (first connection)
100            if !neuron.connections.is_empty() {
101                biases.push(neuron.connections[0].weight);
102
103                // Extract weights (remaining connections)
104                for connection in neuron.connections.iter().skip(1).take(input_size) {
105                    weights.push(connection.weight);
106                }
107            }
108        }
109
110        (weights, biases)
111    }

113    /// Compute gradients for entire batch using GPU-optimized operations
114    /// This method processes the entire batch in parallel on GPU for maximum performance
115    pub fn batch_compute_gradients(
116        &self,
117        network: &Network<T>,
118        all_activations: &Vec<Vec<Vec<T>>>, // [layer][sample][neuron]
119        batch_errors: &[Vec<T>],            // [sample][output_neuron]
120    ) -> Result<(Vec<Vec<T>>, Vec<Vec<T>>), ComputeError> {
121        let batch_size = batch_errors.len();
122        let num_layers = network.layers.len();
123
124        // Initialize gradient accumulators
125        let mut batch_weight_gradients = Vec::new();
126        let mut batch_bias_gradients = Vec::new();
127
128        // Propagate errors backward through the network
129        let mut current_errors = batch_errors.to_vec();
130
131        // Process layers in reverse order
132        for layer_idx in (1..num_layers).rev() {
133            let layer = &network.layers[layer_idx];
134            let prev_layer_size = all_activations[layer_idx - 1][0].len();
135            let current_layer_size = layer.neurons.iter().filter(|n| !n.is_bias).count();
136
137            // Extract layer weights for error propagation
138            let (layer_weights, _) =
139                self.extract_layer_parameters(layer, prev_layer_size, current_layer_size);
140
141            // Compute weight gradients using outer product: gradient = activation^T * error
142            // This can be done efficiently on GPU as a batch matrix multiplication
143            let mut weight_gradients = vec![T::zero(); current_layer_size * prev_layer_size];
144            let mut bias_gradients = vec![T::zero(); current_layer_size];
145
146            // Process entire batch at once
147            for (sample_idx, errors) in current_errors.iter().enumerate() {
148                let prev_activations = &all_activations[layer_idx - 1][sample_idx];
149
150                // Accumulate bias gradients
151                for (i, &error) in errors.iter().enumerate() {
152                    bias_gradients[i] = bias_gradients[i] + error;
153
154                    // Accumulate weight gradients
155                    let weight_offset = i * prev_layer_size;
156                    for (j, &activation) in prev_activations.iter().enumerate() {
157                        weight_gradients[weight_offset + j] =
158                            weight_gradients[weight_offset + j] + error * activation;
159                    }
160                }
161            }
162
163            // Average gradients across batch
164            let batch_size_t = T::from(batch_size).unwrap();
165            for grad in weight_gradients.iter_mut() {
166                *grad = *grad / batch_size_t;
167            }
168            for grad in bias_gradients.iter_mut() {
169                *grad = *grad / batch_size_t;
170            }
171
172            // Store gradients
173            batch_weight_gradients.insert(0, weight_gradients);
174            batch_bias_gradients.insert(0, bias_gradients);
175
176            // Propagate errors to previous layer (if not input layer)
177            if layer_idx > 1 {
178                let mut next_errors = Vec::new();
179
180                for sample_idx in 0..batch_size {
181                    let mut prev_errors = vec![T::zero(); prev_layer_size];
182
183                    // Compute error propagation: prev_error = weights^T * current_error
184                    for (neuron_idx, &error) in current_errors[sample_idx].iter().enumerate() {
185                        let weight_offset = neuron_idx * prev_layer_size;
186                        for prev_idx in 0..prev_layer_size {
187                            prev_errors[prev_idx] = prev_errors[prev_idx]
188                                + layer_weights[weight_offset + prev_idx] * error;
189                        }
190                    }
191
192                    // Apply activation derivative
193                    let prev_layer = &network.layers[layer_idx - 1];
194                    let activation_fn = prev_layer
195                        .neurons
196                        .iter()
197                        .find(|n| !n.is_bias)
198                        .map(|n| n.activation_function)
199                        .unwrap_or(crate::ActivationFunction::Sigmoid);
200
201                    // Apply derivative based on activation function
202                    for (i, &activation) in all_activations[layer_idx - 1][sample_idx]
203                        .iter()
204                        .enumerate()
205                    {
206                        prev_errors[i] = prev_errors[i]
207                            * self.compute_activation_derivative(activation, activation_fn);
208                    }
209
210                    next_errors.push(prev_errors);
211                }
212
213                current_errors = next_errors;
214            }
215        }
216
217        Ok((batch_weight_gradients, batch_bias_gradients))
218    }
219
220    /// Compute activation function derivative
221    fn compute_activation_derivative(
222        &self,
223        output: T,
224        activation_fn: crate::ActivationFunction,
225    ) -> T {
226        use crate::ActivationFunction::*;
227
228        match activation_fn {
229            Sigmoid => output * (T::one() - output), // sigmoid'(x) = sigmoid(x) * (1 - sigmoid(x))
230            Tanh => T::one() - output * output,      // tanh'(x) = 1 - tanhÂ²(x)
231            ReLU => {
232                if output > T::zero() {
233                    T::one()
234                } else {
235                    T::zero()
236                }
237            }
238            Linear => T::one(),
239            _ => output * (T::one() - output), // Default to sigmoid derivative
240        }
241    }

244/// Process forward pass and get activations for all layers using GPU batch operations
245/// This function is optimized to minimize CPU-GPU transfers and maximize GPU shader usage
246pub fn batch_forward_with_activations<
247    T: Float + Send + Sync + Default + std::fmt::Debug + 'static,
248>(
249    network: &Network<T>,
250    batch_inputs: &[Vec<T>],
251    backend: Arc<dyn ComputeBackend<T>>,
252) -> Result<Vec<Vec<Vec<T>>>, ComputeError> {
253    let batch_trainer = BatchGpuTrainer::new(backend, batch_inputs.len());
254
255    // Use the optimized batch forward pass
256    let mut all_layer_activations = vec![batch_inputs.to_vec()]; // Input layer
257    let mut current_activations = batch_inputs.to_vec();
258
259    // Process through each layer using GPU batch operations
260    for (layer_idx, layer) in network.layers.iter().skip(1).enumerate() {
261        let prev_layer_size = current_activations[0].len();
262        let current_layer_size = layer.neurons.iter().filter(|n| !n.is_bias).count();
263
264        // Extract weights and biases once for the entire batch
265        let (weight_matrix, biases) =
266            batch_trainer.extract_layer_parameters(layer, prev_layer_size, current_layer_size);
267
268        // GPU batch matrix multiplication using the batch_matrix_vector_multiply.wgsl shader
269        let batch_outputs = batch_trainer.backend.batch_matrix_vector_multiply(
270            &weight_matrix,
271            &current_activations,
272            current_layer_size,
273            prev_layer_size,
274        )?;
275
276        // Add biases and apply activation function on GPU for the entire batch
277        let mut activated_outputs = Vec::with_capacity(batch_outputs.len());
278        for mut output in batch_outputs {
279            // Add biases
280            for (i, bias) in biases.iter().enumerate() {
281                output[i] = output[i] + *bias;
282            }
283
284            // Apply activation function using GPU activation shaders
285            let activation_fn = layer
286                .neurons
287                .iter()
288                .find(|n| !n.is_bias)
289                .map(|n| n.activation_function)
290                .unwrap_or(crate::ActivationFunction::Sigmoid);
291
292            let activated = batch_trainer.backend.apply_activation_function(
293                &output,
294                activation_fn,
295                T::one(), // steepness
296            )?;
297
298            activated_outputs.push(activated);
299        }
300
301        all_layer_activations.push(activated_outputs.clone());
302        current_activations = activated_outputs;
303    }
304
305    // Convert from [layer][sample][neuron] to [sample][layer][neuron] format
306    let batch_size = batch_inputs.len();
307    let num_layers = all_layer_activations.len();
308    let mut sample_activations = Vec::with_capacity(batch_size);
309
310    for sample_idx in 0..batch_size {
311        let mut sample_acts = Vec::with_capacity(num_layers);
312        for layer_acts in &all_layer_activations {
313            sample_acts.push(layer_acts[sample_idx].clone());
314        }
315        sample_activations.push(sample_acts);
316    }
317
318    Ok(sample_activations)
319}

321/// Optimized batch training step for GPU Adam
322/// This function maximizes GPU shader usage and minimizes CPU-GPU transfers
323pub fn gpu_batch_train_step<T: Float + Send + Sync + Default + std::fmt::Debug + 'static>(
324    network: &mut Network<T>,
325    data: &TrainingData<T>,
326    backend: Arc<dyn ComputeBackend<T>>,
327    adam_params: &mut super::gpu_training::GpuAdam<T>,
328) -> Result<T, ComputeError> {
329    let batch_size = data.inputs.len();
330
331    // Forward pass for entire batch with activations using GPU batch operations
332    let batch_activations = batch_forward_with_activations(network, &data.inputs, backend.clone())?;
333
334    // Get final outputs from the last layer
335    let batch_outputs: Vec<Vec<T>> = batch_activations
336        .iter()
337        .map(|acts| acts.last().unwrap().clone())
338        .collect();
339
340    // Compute output errors and total loss
341    let mut total_error = T::zero();
342    let mut batch_output_errors = Vec::with_capacity(batch_size);
343
344    for (output, target) in batch_outputs.iter().zip(data.outputs.iter()) {
345        let mut sample_errors = Vec::with_capacity(output.len());
346        let mut sample_error = T::zero();
347
348        for (&actual, &desired) in output.iter().zip(target.iter()) {
349            let error = actual - desired;
350            sample_error = sample_error + error * error;
351
352            // Output error derivative for MSE: (actual - desired)
353            sample_errors.push(error);
354        }
355
356        // Divide by number of outputs to match CPU MseError implementation
357        sample_error = sample_error / T::from(output.len()).unwrap();
358        total_error = total_error + sample_error;
359        batch_output_errors.push(sample_errors);
360    }
361
362    // Convert activations to the format expected by batch gradient computation
363    // [sample][layer][neuron] -> [layer][sample][neuron]
364    let num_layers = batch_activations[0].len();
365    let mut layer_activations = Vec::with_capacity(num_layers);
366
367    for layer_idx in 0..num_layers {
368        let mut layer_samples = Vec::with_capacity(batch_size);
369        for sample_acts in &batch_activations {
370            layer_samples.push(sample_acts[layer_idx].clone());
371        }
372        layer_activations.push(layer_samples);
373    }
374
375    // GPU batch gradient computation - processes entire batch efficiently
376    let batch_trainer = BatchGpuTrainer::new(backend, batch_size);
377    let (weight_gradients, bias_gradients) =
378        batch_trainer.batch_compute_gradients(network, &layer_activations, &batch_output_errors)?;
379
380    // Apply Adam parameter updates using computed gradients
381    adam_params.apply_adam_updates_with_gradients(network, &weight_gradients, &bias_gradients)?;
382
383    // Return average error across the batch
384    Ok(total_error / T::from(batch_size).unwrap())
385}
